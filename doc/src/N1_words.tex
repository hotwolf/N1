\section{ANS Forth Words}
\label{N1_words}

\tabref{words:list} provides a list of standard ANS Forth words (see \cite{dpans94})
which map to hardware instructions of the N1 processor.

      \makecell[l]{

      \makecell[l]{

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline
                              
     %0=                                        
      \texttt{0=}                             &

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline
                              
     %1+                                        
      \texttt{1+}                             &
      ( n1$\mid$u1 -- n2$\mid$u2 )            &

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

     %1-                                        
      \texttt{1-}                             &
      ( n1$\mid$u1 -- n2$\mid$u2 )            &

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

     %2*

        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \texttt{2@}                              &

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

        \texttt{0x0000}}                         \\ \hline

    %M*

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      \makecell[l]{

      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

      Bitwise logic OR                       &
      \makecell[l]{                   
        \texttt{0x0000}}                         \\ \hline

        \texttt{0x0000}}                         \\ \hline
